Güvenlik Kuralları

Firebase Console → Firestore → Rules bölümüne aşağıdaki kuralları yapıştırın: 

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAdmin() {
      return request.auth != null;
    }
    match /students/{ogrenciNo} {
      allow read, write: if isAdmin();
      match /veliler/{id}       { allow read, write: if isAdmin(); }
      match /devamsizliklar/{id}{ allow read, write: if isAdmin(); }
      match /davranislar/{id}   { allow read, write: if isAdmin(); }
      match /veligorusmeleri/{id}{ allow read, write: if isAdmin(); }
    }
    match /graduates/{id} {
      allow read, write: if isAdmin();
    }
    match /_autocomplete/{id} {
      allow read, write: if isAdmin();
    }
  }
}
